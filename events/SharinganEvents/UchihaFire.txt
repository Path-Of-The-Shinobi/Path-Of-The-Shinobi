namespace = uchiha_fire

uchiha_fire.0001 = {

	hidden = yes
	
	trigger = {
		has_trait = uchihagene
	}
	
	immediate = {
		if = {
			limit = {
				AND = {
					has_trait = firerelease
					has_trait = uchihagene
				}
			}
			remove_trait = firerelease
			add_trait = firerelease
		}
		else_if = {
			limit = {
				AND = {
					NOT = { has_trait = firerelease }
					has_trait = uchihagene
				}
			}
			add_trait = firerelease
		}
	}
}
